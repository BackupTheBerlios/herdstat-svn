 35 /* base exception */
 36 class herdstat_base_E                   : public std::exception { };
 37 
 38 class herdstat_msg_base_E               : public herdstat_base_E
 39 {
 40     protected:
 41         const char *str;
 42 
 43     public:
 44         herdstat_msg_base_E() { }
 45         herdstat_msg_base_E(const char *msg) : str(msg) { }
 46         herdstat_msg_base_E(const std::string &msg) : str(msg.c_str()) { }
 47         virtual const char *what() const throw() { return str; }
 48 };
 49 
 50 class herdstat_va_base_E                : public herdstat_msg_base_E
 51 {
 52     protected:
 53         const char *str;
 54         va_list v;
 55 
 56     public:
 57         herdstat_va_base_E() { }
 58         herdstat_va_base_E(const char *msg, va_list v)
 59             : str(util::sprintf(msg, v).c_str()) { }
 60         herdstat_va_base_E(const std::string &msg, va_list v)
 61             : str(util::sprintf(msg.c_str(), v).c_str()) { }
 62 
 63         herdstat_va_base_E(const char *msg, ...)
 64         {
 65             va_start(v, msg);
 66             str = util::sprintf(msg, v).c_str();
 67             va_end(v);
 68         }
 69 
 70         herdstat_va_base_E(const std::string &msg, ...)
 71         {
 72             va_start(v, msg.c_str());
 73             str = util::sprintf(msg.c_str(), v).c_str();
 74             va_end(v);
 75         }
 76 
 77         virtual const char *what() const throw() { return str; }
 78 };
