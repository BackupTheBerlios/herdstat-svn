 40 /*
 41  * Default constructor
 42  * Set sane default attributes
 43  */
 44 
 45 format_attrs_T::format_attrs_T()
 46 {
 47     colors = false;
 48     maxtotal = 78;
 49     maxlabel = 20;
 50     maxdata  = 58;
 51 }

 53 /*
 54  * Finalize the attributes ; calculate increase in string length
 55  * because of the colors, etc. This is called once by the caller
 56  * after the attrs have been set.
 57  */
 58 
 59 void
 60 formatter_T::set_attrs()
 61 {
 62     util::color_map_T color;
 63     attr.maxlabel++;
 64     attr.maxtotal = attr.maxlabel + attr.maxdata;
 65 
 66     /* if we're using colors, we need to compensate
 67      * for the increase in string length.           */
 68     if (attr.colors)
 69     {
 70         attr.maxclabel = attr.maxlabel
 71                        + attr.label_color.length()
 72                        + color[none].length();
 73         attr.maxcdata  = attr.maxdata
 74                        + attr.data_color.length()
 75                        + color[none].length();
 76         attr.maxctotal = attr.maxclabel + attr.maxcdata;
 77     }
 78     else
 79     {
 80         attr.label_color = attr.data_color = "";
 81         attr.maxclabel = attr.maxlabel;
 82         attr.maxcdata  = attr.maxdata;
 83         attr.maxctotal = attr.maxtotal;
 84     }
 85 }

 87 /*
 88  * Overloaded append() that takes a vector, converts it to a string,
 89  * and calls the real append()
 90  */
 91 
 92 void
 93 formatter_T::append(const std::string &label, std::vector<std::string> &data)
 94 {
 95     std::string s;
 96     std::vector<std::string>::iterator i;
 97     for (i = data.begin() ; i != data.end() ; ++i)
 98         s += *i + " ";
 99 
100     if (s[s.length() - 1] == ' ')
101         s.erase(s.length() - 1);
102 
103     append(label, s);
104 }

106 /*
107  * Append text to the output buffer.  Smartly handle lines that are bigger
108  * than maxcol (new line and indent), and make sure it all aligns nicely.
109  */
110 
111 void
112 formatter_T::append(const std::string &label, const std::string &data)
113 {
114     util::color_map_T color;
115     std::string cur;
116 
117     if (label.length() > attr.maxlabel)
118         throw format_E("Label '%s' is greater than maxlabel attribute (%d)",
119             label.c_str(), attr.maxlabel);
120 
121     cur = color[green] + label + color[none] +
122         (label.empty() ? "" : ":");
123 
124     while (cur.length() < attr.maxclabel)
125         cur.append(" ");
126 
127     cur.append(attr.data_color);
128 
129     if (not data.empty())
130     {
131         /* will it all fit on one line? */
132         if ((cur.length() + data.length() +
133             color[none].length()) < attr.maxctotal)
134             cur.append(data);
135         else
136         {
137             /* line's full, so find a location where we can truncate */
138             std::string::size_type pos = data.rfind(" ", attr.maxdata);
139             cur.append(pos == std::string::npos ? data : data.substr(0, pos));
140             buffer.push_back(cur);
141             cur.clear();
142 
143             /* indent */
144             while (cur.length() < attr.maxlabel)
145                 cur.append(" ");
146 
147             /* handle leftovers */
148             std::vector<std::string> leftovers = util::splitstr(data.substr(pos));
149 
150             std::vector<std::string>::iterator i;
151             for (i = leftovers.begin() ; i != leftovers.end() ; ++i)
152             {
153                 /* does it fit on the current line? */
154                 if ((cur.length() + (*i).length()) > attr.maxtotal)
155                 {
156                     buffer.push_back(cur);
157                     cur.clear();
158 
159                     /* indent */
160                     while (cur.length() < attr.maxlabel)
161                         cur.append(" ");
162                 }
163 
164                 cur += *i + " ";
165             }
166         }
167     }
168 
169     if (cur.length() > 0)
170         buffer.push_back(cur);
171 }

173 /*
174  * flush everything in the buffer to the specified stream
175  */
176 
177 void
178 formatter_T::flush(std::ostream &stream)
179 {
180     std::vector<std::string>::iterator i;
181     for (i = buffer.begin() ; i != buffer.end() ; ++i)
182         stream << *i << std::endl;
183 
184     buffer.clear();
185 }
