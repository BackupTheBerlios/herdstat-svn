 29 bool
 30 HerdsXMLHandler_T::start_element(const std::string &name, const attrs_type &)
 31 {
 32     if (name == "herd")
 33         in_herd = true;
 34     else if (name == "name" and not in_maintainer)
 35         in_herd_name = true;
 36     else if (name == "email" and not in_maintainer)
 37         in_herd_email = true;
 38     else if (name == "description" and not in_maintainer)
 39         in_herd_desc = true;
 40     else if (name == "maintainer")
 41         in_maintainer = true;
 42     else if (name == "email" and in_maintainer)
 43         in_maintainer_email = true;
 44     else if (name == "name" and in_maintainer)
 45         in_maintainer_name = true;
 46     else if (name == "role")
 47         in_maintainer_role = true;
 48     return true;
 49 }
 50 
 51 bool
 52 HerdsXMLHandler_T::end_element(const std::string &name)
 53 {
 54     if (name == "herd")
 55         in_herd = false;
 56     else if (name == "name" and not in_maintainer)
 57         in_herd_name = false;
 58     else if (name == "email" and not in_maintainer)
 59         in_herd_email = false;
 60     else if (name == "description" and not in_maintainer)
 61         in_herd_desc = false;
 62     else if (name == "maintainer")
 63         in_maintainer = false;
 64     else if (name == "email" and in_maintainer)
 65         in_maintainer_email = false;
 66     else if (name == "name" and in_maintainer)
 67         in_maintainer_name = false;
 68     else if (name == "role")
 69         in_maintainer_role = false;
 70     return true;
 71 }

 73 bool
 74 HerdsXMLHandler_T::text(const std::string &str)
 75 {
 76     if (in_herd_name)
 77     {
 78         cur_herd = str;
 79         herds.insert(std::make_pair(str, new herd_T(str)));
 80     }
 81     else if (in_herd_desc)
 82         herds[cur_herd]->desc = str;
 83     else if (in_maintainer_email)
 84     {
 85         cur_dev = str;
 86         herds[cur_herd]->insert(std::make_pair(str, new dev_attrs_T()));
 87     }
 88     else if (in_maintainer_name)
 89     {
 90         herd_T::iterator i = herds[cur_herd]->find(cur_dev);
 91         if (i == herds[cur_herd]->end())
 92             return false;
 93         i->second->push_back(str);
 94     }
 95     else if (in_maintainer_role)
 96     {
 97         herd_T::iterator i = herds[cur_herd]->find(cur_dev);
 98         if (i == herds[cur_herd]->end())
 99             return false;
100         i->second->push_back(str);
101     }
102 
103     return true;
104 }
105 
106 HerdsXMLHandler_T::~HerdsXMLHandler_T()
107 {
108     for (herds_T::iterator i = herds.begin() ; i != herds.end() ; ++i)
109     {
110         herd_T::iterator h;
111         for (h = i->second->begin() ; h != i->second->end() ; ++h)
112             delete h->second;
113         delete i->second;
114     }
115 }
