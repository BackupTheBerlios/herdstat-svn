 40 /*
 41  * Display a herd and it's developers.
 42  */
 43 
 44 void
 45 herd_T::display(std::ostream &stream)
 46 {
 47     options_T options;
 48     formatter_T out;
 49     util::color_map_T color;
 50 
 51     /* if ECHANGELOG_USER is set try to get the dev email
 52      * so we can highlight it if it occurs in the output  */
 53     std::string user;
 54     std::string::size_type oldlen = 0;
 55     char *result = getenv("ECHANGELOG_USER");
 56     if (result)
 57     {
 58         user = result;
 59         std::string::size_type pos = user.find("<");
 60         if (pos != std::string::npos)
 61             user = user.substr(pos + 1);
 62 
 63         pos = user.find(">");
 64         if (pos != std::string::npos)
 65             user = user.substr(0, pos);
 66     }
 67 
 68     /* sort devs */
 69     std::vector<std::string> sorted_devs;
 70     {
 71         for (iterator i = this->begin() ; i != this->end() ; ++i)
 72         {
 73             std::string dev = i->first;
 74             if (not options.verbose())
 75             {
 76                 /* strip domain names from email addr */
 77                 std::string::size_type pos = dev.find('@');
 78                 if (pos != std::string::npos)
 79                 {
 80                     /* is the current user this user? */
 81                     if (dev == user and not options.quiet())
 82                     {
 83                         /* if so, highlight it ; we need to save maxctotal
 84                          * in order to modify it to compensate for color length */
 85                         oldlen = out.maxctotal();
 86                         dev = color[yellow] + dev.substr(0, pos) + color[none];
 87                     }
 88                     else
 89                         dev = dev.substr(0, pos);
 90                 }
 91             }
 92             sorted_devs.push_back(dev);
 93         }
 94     }
 95     std::sort(sorted_devs.begin(), sorted_devs.end());
 96 
 97     if (not options.quiet())
 98     {
 99         out.append("Herd", name);
100         out.append("Description", desc);
101     }
102 
103     if (options.verbose())
104     {
105         out.append(util::sprintf("Developers(%d)", this->size()), "");
106 
107         std::vector<std::string>::iterator i;
108         for (i = sorted_devs.begin() ; i != sorted_devs.end() ; ++i)
109         {
110             /* display full email; if the user is in the herd, hilite it
111              * to make them feel speshul :) */
112             if (*i == user)
113                 out.append("", color[yellow] + (*i) + color[none]);
114             else
115                 out.append("", color[blue] + (*i) + color[none]);
116 
117             /* display developer attributes (name, role, etc) */
118             iterator x;
119             if ((x = this->find(*i)) != this->end())
120                 x->second->display(std::cout);
121         }
122     }
123     else
124     {
125         if (options.quiet())
126         {
127             std::vector<std::string>::iterator i;
128             for (i = sorted_devs.begin() ; i != sorted_devs.end() ; ++i)
129                 std::cout << *i << std::endl;
130         }
131         else
132         {
133             /* if the current user is in the herd, we highlighted the nick
134              * so adjust maxctotal appropriately                            */
135             if (oldlen != 0)
136                 out.set_maxctotal(oldlen + color[yellow].length() +
137                     color[none].length());
138 
139             out.append(util::sprintf("Developers(%d)", sorted_devs.size()), sorted_devs);
140 
141             /* ...and restore it */
142             if (oldlen != 0)
143                 out.set_maxctotal(oldlen);
144         }
145     }
146 }

148 /*
149  * Display all herds.
150  */
151 
152 void
153 herds_T::display(std::ostream &stream)
154 {
155     options_T options;
156     util::color_map_T color;
157     formatter_T out;
158 
159     std::vector<std::string> hvec;
160     float ndev = 0;
161 
162     if (options.verbose())
163         out.append(util::sprintf("Herds(%d)", this->size()), "");
164 
165     iterator h;
166     size_type n = 0;
167     for (h = this->begin() ; h != this->end() ; ++h)
168         ndev += h->second->size();
169 
170     if (options.verbose())
171         out.append("Avg devs/herd", util::sprintf("%.2f",
172             ndev / this->size()));
173 
174     /* for each herd in herds.xml... */
175     for (h = this->begin() ; h != this->end() ; ++h)
176     {
177         if (options.verbose())
178         {
179             out.append("", color[blue] + h->first + color[none]);
180 
181             iterator x;
182             if ((x = this->find(h->first)) != this->end())
183                 out.append("", x->second->desc);
184 
185             if (++n != this->size())
186                 out.endl();
187         }
188         else
189             hvec.push_back(h->first);
190     }
191 
192     if (not options.verbose())
193     {
194         out.append(util::sprintf("Herds(%d)", this->size()), hvec);
195         out.append("Avg devs/herd", util::sprintf("%.2f", ndev / this->size()));
196     }
197 }
