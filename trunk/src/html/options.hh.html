23 #ifndef HAVE_OPTIONS_HH
24 #define HAVE_OPTIONS_HH 1

26 #ifdef HAVE_CONFIG_H
27 # include "config.h"

30 #include <string>
31 
32 enum options_action_T
33 {
34     action_unspecified,
35     action_herd,
36     action_dev,
37     action_pkg
38 };
39 
40 class options_T
41 {
42     protected:
43         static std::string _herds_xml;
44         static std::string _portdir;
45         static bool _debug;
46         static bool _timer;
47         static bool _verbose;
48         static bool _quiet;
49         static std::string::size_type _maxcol;
50         static options_action_T _action;
51 
52     public:
53         void set_debug(bool value) { options_T::_debug = value; }
54         bool debug() { return options_T::_debug; }
55 
56         void set_timer(bool value) { options_T::_timer = value; }
57         bool timer() { return options_T::_timer; }
58 
59         void set_verbose(bool value) { options_T::_verbose = value; }
60         bool verbose() { return options_T::_verbose; }
61 
62         void set_quiet(bool value) { options_T::_quiet = value; }
63         bool quiet() { return options_T::_quiet; }
64 
65         void set_maxcol(std::string::size_type value) { options_T::_maxcol = value - 2; }
66         std::string::size_type maxcol() { return options_T::_maxcol; }
67 
68         void set_action(options_action_T value) { options_T::_action = value; }
69         options_action_T action() { return options_T::_action; }
70 
71         void set_herds_xml(std::string const &value) { options_T::_herds_xml = value; }
72         std::string const &herds_xml() { return options_T::_herds_xml; }
73 
74         void set_portdir(std::string const &value) { options_T::_portdir = value; }
75         std::string const &portdir() { return options_T::_portdir; }
76 };
77 
78 #endif
