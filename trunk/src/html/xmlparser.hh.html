23 #ifndef HAVE_PARSER_HH

28 #endif

30 #include <istream>
31 #include <string>
32 #include <xmlwrapp/init.h>
33 #include <xmlwrapp/event_parser.h>
34 #include "exceptions.hh"
35 
36 class XMLHandler_T : public xml::event_parser
37 {
38     protected:
39         virtual bool start_element(const std::string &, const attrs_type &) = 0;
40         virtual bool end_element(const std::string &) = 0;
41         virtual bool text(const std::string &) = 0;
42 };
43 
44 class XMLParser_T
45 {
46     protected:
47         xml::init xmlinit;
48         XMLHandler_T *handler;
49 
50     public:
51         XMLParser_T(XMLHandler_T *h) : handler(h)
52         {
53             xmlinit.remove_whitespace(true);
54         }
55 
56         virtual ~XMLParser_T() { }
57 
58         virtual void parse(const std::string &f)
59         {
60             if (not handler->parse_file(f.c_str()))
61                 throw XMLParser_E(f, handler->get_error_message());
62         }
63 
64         virtual void parse(std::istream &stream)
65         {
66             if (not handler->parse_stream(stream))
67                 throw XMLParser_E("stream", handler->get_error_message());
68         }
69 };
70 
71 #endif
