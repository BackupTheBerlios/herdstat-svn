 76     class timer_T
 77     {
 78         private:
 79             struct timeval _begin, _end;
 80             long ms;
 81         public:
 82             void start() { gettimeofday(&_begin, NULL); }
 83             void stop()
 84             {
 85                 gettimeofday(&_end, NULL);
 86                 ms = _end.tv_sec - _begin.tv_sec;
 87                 ms *= 1000;
 88                 ms += (_end.tv_usec - _begin.tv_usec) / 1000;
 89             }
 90             long elapsed() const { return ms; }
 91     };
