 40 /* format attributes */
 41 class format_attrs_T
 42 {
 43     public:
 44         format_attrs_T();
 45 
 46         bool colors;
 47         std::string label_color;
 48         std::string data_color;
 49 
 50         std::string::size_type maxtotal;
 51         std::string::size_type maxlabel;
 52         std::string::size_type maxdata;
 53 
 54         std::string::size_type maxctotal;
 55         std::string::size_type maxclabel;
 56         std::string::size_type maxcdata;
 57 };
 58 
 59 class formatter_T
 60 {
 61     private:
 62         static std::vector<std::string> buffer;
 63         static format_attrs_T attr;
 64 
 65     public:
 66         typedef int size_type;
 67 
 68         formatter_T() { }
 69 
 70         void endl() { buffer.push_back(""); }
 71         void append(const std::string &, const std::string &);
 72         void append(const std::string &, std::vector<std::string> &);
 73         void flush(std::ostream &);
 74 
 75         void set_attrs();
 76 
 77         /* attribute member functions */
 78         void set_maxtotal(size_type s) { attr.maxtotal = s; }
 79         size_type maxtotal() { return attr.maxtotal; }
 80         void set_maxctotal(size_type s) { attr.maxctotal = s; }
 81         size_type maxctotal() { return attr.maxctotal; }
 82 
 83         void set_maxlabel(size_type s) { attr.maxlabel = s; }
 84         size_type maxlabel() { return attr.maxlabel; }
 85         void set_maxclabel(size_type s) { attr.maxclabel = s; }
 86         size_type maxclabel() { return attr.maxclabel; }
 87 
 88         void set_maxdata(size_type s) { attr.maxdata = s; }
 89         size_type maxdata() { return attr.maxdata; }
 90         void set_maxcdata(size_type s) { attr.maxcdata = s; }
 91         size_type maxcdata()  { return attr.maxcdata;  }
 92 
 93         void set_colors(bool value) { attr.colors = value; }
 94         bool colors() { return attr.colors; }
 95 
 96         void set_labelcolor(std::string &s) { attr.label_color = s; }
 97         std::string &labelcolor() { return attr.label_color; }
 98 
 99         void set_datacolor(std::string &s) { attr.data_color = s; }
100         std::string &datacolor() { return attr.data_color; }
101 };
