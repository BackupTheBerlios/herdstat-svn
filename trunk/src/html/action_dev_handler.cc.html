 39 int
 40 action_dev_handler_T::operator() (herds_T &herds_xml,
 41                                   std::vector<std::string> &devs)
 42 {
 43     options_T options;
 44     util::color_map_T color;
 45 
 46     /* set format attributes */
 47     formatter_T output;
 48     output.set_maxlabel(10);
 49     output.set_maxdata(options.maxcol() - output.maxlabel());
 50     output.set_colors(true);
 51     output.set_labelcolor(color[green]);
 52     output.set_attrs();
 53 
 54     /* for each specified dev... */
 55     std::vector<std::string>::iterator dev;
 56     std::vector<std::string>::size_type n = 0;
 57     for (dev = devs.begin() ; dev != devs.end() ; ++dev)
 58     {
 59         std::vector<std::string> herds;
 60 
 61         /* for each herd in herds.xml... */
 62         herds_T::iterator h;
 63         herds_T::size_type nherd = 0;
 64         for (h = herds_xml.begin() ; h != herds_xml.end() ; ++h)
 65         {
 66             std::string herd = h->first;
 67 
 68             /* is the dev in the current herd? */
 69             herd_T::iterator d = herds_xml[herd]->find(*dev + "@gentoo.org");
 70             if (d != herds_xml[herd]->end())
 71                 herds.push_back(herd);
 72             else
 73             {
 74                 /* there have been cases where the dev put just their
 75                  * nick instead of their full email address             */
 76                 d = herds_xml[herd]->find(*dev);
 77                 if (d != herds_xml[herd]->end())
 78                     herds.push_back(herd);
 79             }
 80         }
 81 
 82         /* was the dev in any of the herds? */
 83         if (herds.empty())
 84         {
 85             std::cerr << color[red] << "'" << *dev << "' doesn't seem to "
 86                 << "belong to any herds." << color[none] << std::endl;
 87 
 88             if (devs.size() == 1)
 89                 throw dev_E();
 90         }
 91         else
 92         {
 93             if (not options.quiet())
 94             {
 95                 output.append("Developer", *dev);
 96 
 97                 if (options.verbose())
 98                 {
 99                     output.append(util::sprintf("Herds(%d)", herds.size()), "");
100 
101                     std::vector<std::string>::iterator i;
102                     for (i = herds.begin() ; i != herds.end() ; ++i)
103                     {
104                         /* display herd */
105                         output.append("", color[blue] + (*i) + color[none]);
106                         /* display herd description */
107                         output.append("", herds_xml[*i]->desc);
108 
109                         if (++nherd != herds.size())
110                             output.endl();
111                     }
112                 }
113                 else
114                 {
115                     output.append(util::sprintf("Herds(%d)", herds.size()),
116                         herds);
117                 }
118             }
119             else
120             {
121                 std::vector<std::string>::iterator i;
122                 for (i = herds.begin() ; i != herds.end() ; ++i)
123                     std::cout << *i << std::endl;
124             }
125         }
126 
127         /* skip a line if we're not displaying the last one */
128         if (++n != devs.size())
129         {
130             if (options.quiet())
131                 std::cout << std::endl;
132             else
133                 output.endl();
134         }
135     }
136 
137     if (not options.quiet())
138         output.flush(std::cout);
139 
140     return EXIT_SUCCESS;
141 }
