 51 /*
 52  * Retrieve a list of all categories from PORTDIR/profiles/categories
 53  */
 54 
 55 std::vector<std::string>
 56 get_categories(const std::string &portdir)
 57 {
 58     std::string catfile = portdir + "/profiles/categories";
 59     std::vector<std::string> categories;
 60     std::auto_ptr<std::istream> f(new std::ifstream(catfile.c_str()));
 61 
 62     if ((*f))
 63     {
 64         while (not f->eof())
 65         {
 66             std::string s;
 67             if (std::getline(*f, s))
 68                 categories.push_back(s);
 69         }
 70     }
 71     else
 72         throw bad_fileobject_E("Failed to open '%s': %s", catfile.c_str(),
 73             strerror(errno));
 74 
 75     return categories;
 76 }

 78 /*
 79  * Return a list of _all_ metadata.xml files.
 80  */
 81 
 82 std::vector<std::string>
 83 get_metadatas(const std::string &portdir)
 84 {
 85     std::vector<std::string> metadatas;
 86     std::vector<std::string> categories = get_categories(portdir);
 87     std::vector<std::string>::iterator c;
 88 
 89     for (c = categories.begin() ; c != categories.end() ; ++c)
 90     {
 91         std::string cat = portdir + "/" + (*c);
 92 
 93         /* make sure it's valid before trying to open it.
 94          * unfortunately profiles/categories isn't always 100% up-to-date.  */
 95         if (not util::is_dir(cat))
 96             continue;
 97 
 98         /* open category */
 99         DIR *dir = opendir(cat.c_str());
100         if (not dir)
101         {
102             throw bad_fileobject_E("Failed to open directory '%s': %s",
103                 cat.c_str(), strerror(errno));
104         }
105 
106         util::debug_msg("opened directory %s", cat.c_str());
107 
108         struct dirent *d;
109         while ((d = readdir(dir)))
110         {
111             std::string metadata = cat + "/" + d->d_name + "/metadata.xml";
112             if (util::is_file(metadata))
113                 metadatas.push_back(metadata);
114         }
115 
116         closedir(dir);
117     }
118 
119     return metadatas;
120 }

123 /*
124  * Given a list of herds, determine all packages in each herd.
125  * For reliability reasons, every metadata.xml in the tree is
126  * parsed.
127  */
128 
129 int
130 action_pkg_handler_T::operator() (herds_T &herds_xml,
131                                   std::vector<std::string> &herds)
132 {
133     options_T options;
134     util::color_map_T color;
135     util::timer_T timer;
136 
137     /* set format attributes */
138     formatter_T output;
139     output.set_maxlabel(16);
140     output.set_maxdata(options.maxcol() - output.maxlabel());
141     output.set_colors(true);
142     output.set_labelcolor(color[green]);
143     output.set_attrs();
144 
145     /* get a list of every metadata.xml */
146     std::vector<std::string> metadatas;
147     try
148     {
149         /* PORTDIR */
150         options.set_portdir(util::portdir());
151         char *result = getenv("PORTDIR");
152         if (result)
153             options.set_portdir(result);
154 
155         /* make sure it exists */
156         if (not util::is_dir(options.portdir()))
157             throw bad_fileobject_E("PORTDIR '%s' does not exist.",
158                 options.portdir().c_str());
159 
160         /* get a list of all metadata.xml's */
161         metadatas = get_metadatas(options.portdir());
162     }
163     catch (const bad_fileobject_E &e)
164     {
165         std::cerr << color[red] << e.what() << color[none] << std::endl;
166         return EXIT_FAILURE;
167     }
168 
169     /* for each specified herd... */
170     std::vector<std::string>::iterator herd;
171     std::vector<std::string>::size_type n = 0;
172     for (herd = herds.begin() ; herd != herds.end() ; ++herd)
173     {
174         std::map<std::string, std::string> pkgs;
175 
176         /* does the herd exist? */
177         if (not herds_xml.exists(*herd))
178         {
179             std::cerr << color[red] << "Herd '" << *herd
180                 << "' doesn't seem to exist." << color[none] << std::endl;
181 
182             /* if the user specified more than one herd, then just print
183              * the error and keep going; otherwise, we want to exit with
184              * an error code
185              */
186             if (herds.size() > 1)
187             {
188                 std::cerr << std::endl;
189                 continue;
190             }
191             else
192                 throw herd_E();
193         }
194 
195         if (options.timer())
196             timer.start();
197 
198         /* for each metadata.xml... */
199         std::vector<std::string>::iterator m;
200         for (m = metadatas.begin() ; m != metadatas.end() ; ++m)
201         {
202             try
203             {
204                 std::auto_ptr<MetadataXMLHandler_T> mhandler(new
205                     MetadataXMLHandler_T());
206                 XMLParser_T parser(&(*mhandler));
207 
208                 /* parse metadata.xml */
209                 parser.parse(*m);
210 
211                 /* search the metadata for our herd */
212                 std::vector<std::string>::iterator pos;
213                 pos = std::find(mhandler->herds.begin(),
214                     mhandler->herds.end(), *herd);
215 
216                 /* is it there? */
217                 if (pos == mhandler->herds.end())
218                     continue;
219                 else
220                 {
221                     /* get category/package from absolute path */
222                     std::string cat_and_pkg = (*m).substr(options.portdir().size() + 1);
223                     std::string::size_type pos = cat_and_pkg.find("/metadata.xml");
224                     if (pos != std::string::npos)
225                         cat_and_pkg = cat_and_pkg.substr(0, pos);
226 
227                     util::debug_msg("Match found in %s.", (*m).c_str());
228 
229                     pkgs.insert(std::make_pair(cat_and_pkg, mhandler->longdesc));
230                 }
231             }
232             catch (const XMLParser_E &e)
233             {
234                 std::cerr << color[red] << "Error parsing '" << e.file()
235                     << "': " << e.error() << color[none] << std::endl;
236                 return EXIT_FAILURE;
237             }
238         }
239 
240         if (options.timer())
241             timer.stop();
242 
243         /* we now have the list of packages that correspond to the herd */
244 
245         if (not options.quiet())
246         {
247             output.append("Herd", *herd);
248             output.append("Description", herds_xml[*herd]->desc);
249             output.append(util::sprintf("Packages(%d)", pkgs.size()), "");
250         }
251 
252         /* display the category/package */
253         std::map<std::string, std::string>::iterator p;
254         for (p = pkgs.begin() ; p != pkgs.end() ; ++p)
255         {
256             if (options.quiet())
257                 std::cout << p->first << std::endl;
258             else
259             {
260                 /* TODO: the below code works, but until we figure out a way
261                  * to cleanup all the whitespace, the output looks like shit */
262 
263 //                if (options.verbose() and not p->second.empty())
264 //                {
265 //                    output.append("", color[blue] + p->first + color[none]);
266 //                    output.append("", p->second);
267 //                }
268 //                else
269                     output.append("", p->first);
270             }
271         }
272 
273         /* only skip a line if we're not on the last one */
274         if (++n != herds.size())
275             output.endl();
276     }
277 
278     if (not options.quiet())
279         output.flush(std::cout);
280 
281     if (options.timer())
282     {
283         std::cout << std::endl << "Took " << timer.elapsed()
284             << "ms to parse every metadata.xml in the tree." << std::endl;
285     }
286 
287     return EXIT_SUCCESS;
288 }
