 44 int
 45 action_herd_handler_T::operator() (herds_T &herds_xml,
 46                                    std::vector<std::string> &herds)
 47 {
 48     options_T options;
 49     util::color_map_T color;
 50 
 51     /* set format attributes */
 52     formatter_T output;
 53     output.set_maxlabel(15);
 54     output.set_maxdata(options.maxcol() - output.maxlabel());
 55     output.set_colors(true);
 56     output.set_labelcolor(color[green]);
 57     output.set_attrs();
 58 
 59     /* was the all target specified? */
 60     if (herds[0] == "all")
 61         herds_xml.display(std::cout);
 62 
 63     /* nope, so only display stats for specified herds */
 64     else
 65     {
 66         /* for each specified herd... */
 67         std::vector<std::string>::iterator herd;
 68         std::vector<std::string>::size_type n = 0;
 69         for (herd = herds.begin() ; herd != herds.end() ; ++herd)
 70         {
 71             /* does the herd exist? */
 72             if (not herds_xml.exists(*herd))
 73             {
 74                 std::cerr << color[red] << "Herd '" << *herd
 75                     << "' doesn't seem to exist." << color[none] << std::endl;
 76 
 77                 /* if the user specified more than one herd, then just print
 78                  * the error and keep going; otherwise, we want to exit with
 79                  * an error code */
 80                 if (herds.size() > 1)
 81                 {
 82                     std::cerr << std::endl;
 83                     continue;
 84                 }
 85                 else
 86                     throw herd_E();
 87             }
 88 
 89             herds_xml[*herd]->display(std::cout);
 90 
 91             /* only skip a line if we're not displaying the last one */
 92             if (++n != herds.size())
 93                 output.endl();
 94         }
 95     }
 96 
 97     if (not options.quiet())
 98         output.flush(std::cout);
 99 
100     if (options.timer())
101         std::cout << std::endl;
102 
103     return EXIT_SUCCESS;
104 }
