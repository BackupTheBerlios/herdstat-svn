 47 /*
 48  * Return the output of the specified command
 49  */
 50 
 51 std::string
 52 util::get_output_of(const char *cmd)
 53 {
 54     std::string output;
 55     FILE *p = popen(cmd, "r");
 56     if (p)
 57     {
 58         char line[PATH_MAX+1];
 59         if (std::fgets(line, sizeof(line) - 1, p) != NULL)
 60             output = line;
 61         pclose(p);
 62     }
 63 
 64     return output;
 65 }

 67 /*
 68  * Determine PORTDIR
 69  */
 70 
 71 const char *
 72 util::portdir()
 73 {
 74     const char *make_conf = "/etc/make.conf";
 75     const char *make_glob = "/etc/make.globals";
 76 
 77     std::string portdir;
 78 
 79     if (util::is_file(make_conf))
 80     {
 81         std::auto_ptr<std::ifstream> f(new std::ifstream(make_conf));
 82         if (not (*f))
 83             throw bad_fileobject_E("%s: %s", make_conf, strerror(errno));
 84 
 85         util::rcfile_T rc(*f);
 86         util::rcfile_keys_T::iterator pos = rc.keys.find("PORTDIR");
 87         if (pos != rc.keys.end())
 88             portdir = pos->second;
 89     }
 90 
 91     if (portdir.empty() and util::is_file(make_glob))
 92     {
 93         std::auto_ptr<std::ifstream> f(new std::ifstream(make_glob));
 94         if (not (*f))
 95             throw bad_fileobject_E("%s: %s", make_glob, strerror(errno));
 96 
 97         util::rcfile_T rc(*f);
 98         util::rcfile_keys_T::iterator pos = rc.keys.find("PORTDIR");
 99         if (pos != rc.keys.end())
100             portdir = pos->second;
101     }
102 
103     return (portdir.empty() ? "/usr/portage" : portdir.c_str());
104 }

116 int
117 util::fetch(const char *url, const char *file)
118 {
119     const char *dir = util::dirname(file);
120 
121     /* we have permission to write */
122     if (access(dir, W_OK) != 0)
123         throw bad_fileobject_E("%s: %s", dir, strerror(errno));
124 
125     std::string cmd = util::sprintf("%s -rq -t3 -T5 -O %s '%s'", WGET, file, url);
126 
127     util::debug_msg("Executing '%s'", cmd.c_str());
128 
129     return std::system(cmd.c_str());
130 }

133 /*
134  * Is the specified path a directory?
135  */
136 bool
137 util::is_dir(const char *path)
138 {
139     struct stat s;
140     if (stat(path, &s) != 0)
141         return false;
142     return S_ISDIR(s.st_mode);
143 }

178 const char *
179 util::basename(const char *path)
180 {
181     std::string s = path;
182 
183     /* chop any trailing /'s */
184     while (s[s.length() - 1] == '/' and s.length() > 1)
185         s.erase(s.length() - 1);
186 
187     char *p = std::strrchr(s.c_str(), '/');
188     return (p ? (*(p + 1) != '\0' ? p + 1 : p) : path);
189 }
190 
191 const char *
192 util::basename(std::string const &path)
193 {
194     return util::basename(path.c_str());
195 }

197 const char *
198 util::dirname(const char *path)
199 {
200     std::string s = path;
201     size_t path_len = strlen(path);
202     size_t base_len = strlen(util::basename(path));
203     size_t len = (path[path_len - 1] == '/' ? 2 : 1);
204     len = path_len - base_len - (path_len - base_len - len == 0 ? len - 1 : len);
205 
206     return s.substr(0, len).c_str();
207 }

215 /*
216  * Return a string formatted with printf-like format specifier
217  */
218 std::string
219 util::sprintf(const char *str, ...)
220 {
221     va_list v;
222     va_start(v, str);
223     std::string s(util::sprintf(str, v));
224     va_end(v);
225     return s;
226 }

228 /*
229  * Overloaded sprintf that takes a string and va_list
230  */
231 std::string
232 util::sprintf(const char *str, va_list v)
233 {
234 #ifdef HAVE_VASPRINTF
235     char *buf;
236     vasprintf(&buf, str, v);
237 #else
238     char buf[4096] = { 0 };
239 # ifdef HAVE_VSNPRINTF
240     vsnprintf(buf, sizeof(buf), str, v);
241 # else
242     vsprintf(buf, str, v);
243 # endif
244 #endif
245 
246     std::string s(buf);
247 
248 #ifdef HAVE_VASPRINTF
249     free(buf);
250 #endif
251 
252     return s;
253 }

255 /*
256  * Given a string and a delimiter, split the string,
257  * returning all the substrings in a vector
258  */
259 std::vector<std::string>
260 util::splitstr(const std::string &str, const char delim)
261 {
262     std::vector<std::string> vec;
263     std::string::size_type pos, lpos = 0;
264 
265     while (true)
266     {
267         pos = str.find(delim, lpos);
268         if (pos == std::string::npos)
269         {
270             vec.push_back(str.substr(lpos));
271             break;
272         }
273 
274         /* don't append empty strings (two
275          * delimiters in a row were encountered) */
276         if (str.substr(lpos, pos - lpos).length() > 0)
277             vec.push_back(str.substr(lpos, pos - lpos));
278 
279         lpos = ++pos;
280     }
281     return vec;
282 }

315 util::rcfile_T::rcfile_T(std::ifstream &stream)
316 {
317     std::string line;
318     std::string::size_type pos;
319 
320     while (std::getline(stream, line))
321     {
322         pos = line.find_first_not_of(" \t");
323         if (pos != std::string::npos)
324             line.erase(0, pos);
325 
326         if (line.length() < 1 or line[0] == '#')
327             continue;
328 
329         pos = line.find('=');
330         if (pos != std::string::npos)
331         {
332             std::string key = line.substr(0, pos);
333             std::string val = line.substr(pos + 1);
334 
335             if (std::string::npos != (pos = key.find_first_not_of(" \t")))
336                 key.erase(0, pos);
337             if (std::string::npos != (pos = val.find_first_not_of(" \t")))
338                 val.erase(0, pos);
339             if (std::string::npos != (pos = key.find_last_not_of(" \t")))
340                 key.erase(++pos);
341             if (std::string::npos != (pos = val.find_last_not_of(" \t")))
342                 val.erase(++pos);
343 
344             if (std::string::npos != (pos = val.find_first_of("'\"")))
345             {
346                 val.erase(pos, pos + 1);
347                 if (std::string::npos != (pos = val.find_last_of("'\"")))
348                     val.erase(pos, pos + 1);
349             }
350 
351             keys[key] = val;
352         }
353     }
