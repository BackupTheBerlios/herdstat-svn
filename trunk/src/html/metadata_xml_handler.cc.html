30 bool
31 MetadataXMLHandler_T::start_element(const std::string &name, const attrs_type &)
32 {
33     if (name == "herd")
34         in_herd = true;
35     else if (name == "maintainer")
36         in_maintainer = true;
37     else if (name == "email")
38         in_email = true;
39     else if (name == "name")
40         in_name = true;
41     else if (name == "description")
42         in_desc = true;
43     else if (name == "longdescription")
44         in_longdesc = true;
45 
46     return true;
47 }
48 
49 bool
50 MetadataXMLHandler_T::end_element(const std::string &name)
51 {
52     if (name == "herd")
53         in_herd = false;
54     else if (name == "maintainer")
55         in_maintainer = false;
56     else if (name == "email")
57         in_email = false;
58     else if (name == "name")
59         in_name = false;
60     else if (name == "description")
61         in_desc = false;
62     else if (name == "longdescription")
63         in_longdesc = false;
64 
65     return true;
66 }
67 
68 bool
69 MetadataXMLHandler_T::text(const std::string &str)
70 {
71     if (in_herd)
72         herds.push_back(str);
73     else if (in_email)
74     {
75         cur_dev = str;
76         devs[str] = new dev_attrs_T();
77     }
78     else if (in_name or in_desc)
79         devs[cur_dev]->push_back(str);
80     else if (in_longdesc)
81         longdesc = str;
82 
83     return true;
84 }
85 
86 MetadataXMLHandler_T::~MetadataXMLHandler_T()
87 {
88     herd_T::iterator i;
89     for (i = devs.begin() ; i != devs.end() ; ++i)
90         delete i->second;
91 }
